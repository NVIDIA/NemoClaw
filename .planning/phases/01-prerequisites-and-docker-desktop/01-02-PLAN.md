The test file structure:

1. **Setup block** -- dot-source install.ps1 to import functions, but prevent the entry-point code from executing. Do this by wrapping the dot-source in a way that mocks `Assert-Administrator` and `Install-Prerequisites` call at script level. Simplest approach: extract functions into a separate module or use `BeforeAll` with selective mocking.

Recommended approach -- since install.ps1 has `Assert-Administrator` and `Install-Prerequisites` at the bottom as entry-point calls, the test file should:
```powershell
BeforeAll {
    # Prevent entry-point execution by mocking the functions that run at script scope
    # We'll dot-source after setting up the mock context
    $scriptPath = "$PSScriptRoot\..\install.ps1"

    # Create a modified version that comments out the entry point for testing
    $scriptContent = Get-Content $scriptPath -Raw
    # Remove the last two executable lines (Assert-Administrator / Install-Prerequisites)
    # by only dot-sourcing function definitions
    # Alternative: just mock exit and admin check
    Mock exit {} -ModuleName *
    Mock Start-Process {} -ModuleName *
    . $scriptPath
}
```

Actually, the cleanest approach: Add a guard at the bottom of install.ps1 in Plan 01 Task 3. But since we can't modify plan 01 now, handle it in the test:

Describe "Docker Desktop Installation" -Tag "DockerInstall", "PREREQ-08" {
    It "Install-DockerDesktop function exists" {
        Get-Command Install-DockerDesktop | Should -Not -BeNullOrEmpty
    }
    It "skips installation when Docker Desktop already exists" {
        Mock Test-Path { $true } -ParameterFilter { $Path -like "*Docker Desktop*" }
        Mock winget {}
        Install-DockerDesktop
        Should -Invoke winget -Times 0
    }
    It "tries winget first when available" {
        Mock Test-Path { $false } -ParameterFilter { $Path -like "*Docker Desktop*" }
        Mock Get-Command { $true } -ParameterFilter { $Name -eq "winget" }
        Mock winget { $global:LASTEXITCODE = 0 }
        Install-DockerDesktop
        Should -Invoke winget -Times 1
    }
    It "falls back to EXE download when winget unavailable" {
        Mock Test-Path { $false } -ParameterFilter { $Path -like "*Docker Desktop*" }
        Mock Get-Command { $null } -ParameterFilter { $Name -eq "winget" }
        Mock Install-DockerDesktopFromExe {}
        Install-DockerDesktop
        Should -Invoke Install-DockerDesktopFromExe -Times 1
    }
    It "uses --backend=wsl-2 for EXE installer" {
        $content = Get-Content "$PSScriptRoot\..\install.ps1" -Raw
        $content | Should -Match "--backend=wsl-2"
    }
}
